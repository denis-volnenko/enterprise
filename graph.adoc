= Структуры JSON-объектов

== Плоские структуры

*ProjectPlainDto*
----
{
  "id": "1",
  "name": "Демо проект"
}
----

*TaskPlainDto*
----
{
  "id": "1",
  "name": "Демо задача",
  "projectId": "1"
}
----

== Граф (нециклический, направленный, полный)

*ProjectGraphDto*
----
{
  "id": "1",
  "name": "Демо проект",
  "tasks": [{
    "id": "1",
    "name": "Первая задача"
  },{
    "id": "2",
    "name": "Вторая задача"
  }]
}
----

*TaskGraphDto*
----
{
  "id": "1",
  "name": "Первая задача",
  "project": {
    "id": "1",
    "name": "Демо проект"
  }
}
----

== Граф (нециклический, направленный, облегченный)

*ProjectGraphDto*
----
{
  "id": "1",
  "name": "Демо проект",
  "tasks": ["1", "2"]
}
----

----
{
  "id": "1",
  "name": "Первая задача",
  "project": "1"
}
----
